```rust
//! GPU encoder detection for hardware-accelerated FFmpeg encoding.
//!
//! Detects available GPU encoders via `ffmpeg -encoders` at startup and
//! provides encoder name injection and CPU fallback logic.

use std::process::Command;

use crate::models::gpu::GpuEncoder;

/// Detect best available GPU encoder for this platform.
/// Returns None if no hardware encoder found (CPU fallback to libx264).
pub fn detect_gpu_encoder(ffmpeg_path: &str) -> Option<GpuEncoder> {
    let ffmpeg_bin = std::path::Path::new(ffmpeg_path)
        .join(if cfg!(target_os = "windows") {
            "ffmpeg.exe"
        } else {
            "ffmpeg"
        });

    let output = Command::new(&ffmpeg_bin)
        .args(["-hide_banner", "-encoders"])
        .output()
        .ok()?;

    let stdout = String::from_utf8_lossy(&output.stdout);

    #[cfg(target_os = "macos")]
    {
        if stdout.contains("h264_videotoolbox") {
            return Some(GpuEncoder::VideoToolbox);
        }
    }
    #[cfg(target_os = "windows")]
    {
        if stdout.contains("h264_nvenc") {
            return Some(GpuEncoder::Nvenc);
        }
        if stdout.contains("h264_amf") {
            return Some(GpuEncoder::Amf);
        }
    }
    #[cfg(target_os = "linux")]
    {
        if stdout.contains("h264_vaapi") {
            return Some(GpuEncoder::Vaapi);
        }
    }

    None
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_detect_gpu_encoder_no_ffmpeg() {
        let result = detect_gpu_encoder("/nonexistent/path");
        assert!(result.is_none());
    }
}
```

Per D-04: detection is platform-specific via `cfg!(target_os)` guards. Encoder names match RESEARCH.md confirmed names: `h264_videotoolbox`, `h264_nvenc`, `h264_amf`, `h264_vaapi`.
Per D-05: returns `None` if no encoder found (caller falls back to CPU).
Per RESEARCH Pitfall 3: detection only checks encoder listing, not actual hardware availability (handled by retry-on-fail in executor).

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-03-01 | Tampering | gpu.rs detect_gpu_encoder() | accept | FFmpeg stdout is trusted (it is the locally installed binary). `from_utf8_lossy` handles non-UTF8 bytes safely (replaces with replacement chars). `.ok()?` short-circuits on process failure. |
| T-05-03-02 | Denial of Service | GPU encoder process hang | mitigate | Per D-05: GPU encode failure triggers retry with CPU encoder. The retry is handled in Plan 05-05/05-06 batch loop (catch Err, retry with gpu_encoder=None). The detection itself is a one-shot `output()` call with implicit OS timeout. |
| T-05-03-03 | Elevation of Privilege | GPU encoder name injection into CLI | accept | Encoder names are hardcoded in `GpuEncoder::encoder_name()` (match arms return string literals). No user-supplied strings reach the `-c:v` argument. No injection vector. |
| T-05-03-04 | Information Disclosure | gpu-encoder-detected event | accept | The event payload contains only the `GpuEncoder` enum variant name (e.g., "Nvenc"). No system paths, driver versions, or hardware details are emitted. |
